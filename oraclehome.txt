CREATE OR REPLACE PROCEDURE STRING_SPLIT_PRC(P_STRING IN VARCHAR2)
AS
  V_STRING_LENGTH NUMBER := LENGTH(P_STRING);
  V_CHAR          VARCHAR2(1);
  V_NUM           NUMBER;
BEGIN
  FOR I IN 1..V_STRING_LENGTH
  LOOP
    V_CHAR    := SUBSTR(P_STRING,I,1);
    IF V_CHAR != ',' THEN
      DBMS_OUTPUT.PUT_LINE(V_CHAR);
    END IF;
  END LOOP;
EXCEPTION
WHEN OTHERS THEN
  DBMS_OUTPUT.PUT_LINE(SQLERRM);
END;
